def initialize(base_uri)
  #sets the ip:port/base_route
  @base_uri = case base_uri
                when Hash
                  URI::HTTP.build(base_uri).to_s
                when URI::HTTP
                  base_uri.to_s
                when String
                  base_uri
                else
                  raise ArgumentError.new "Invalid URI class: #{base_uri.class}"
              end
end
